   238→   238→function LoginDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   239→   239→  const { locale, setUser } = useAppStore();
   240→   240→  const [email, setEmail] = useState('');
   241→   241→  const [password, setPassword] = useState('');
   242→   242→  const [showPassword, setShowPassword] = useState(false);
   243→   243→  const [rememberMe, setRememberMe] = useState(false);
   244→   244→  const [loading, setLoading] = useState(false);
   245→   245→  const [error, setError] = useState('');
   246→   246→
   247→   247→  const handleSubmit = async (e: React.FormEvent) => {
   248→   248→    e.preventDefault();
   249→   249→    setError('');
   250→   250→    setLoading(true);
   251→   251→    try {
   252→   252→      const res = await fetch('/api/v1/auth/login', {
   253→   253→        method: 'POST',
   254→   254→        headers: { 'Content-Type': 'application/json' },
   255→   255→        body: JSON.stringify({ email, password, rememberMe }),
   256→   256→      });
   257→   257→      const data = await res.json();
   258→   258→      if (!data.success) {
   259→   259→        setError(data.error || t('common.error', locale));
   260→   260→        return;
   261→   261→      }
   262→   262→      setUser(
   263→   263→        {
   264→   264→          id: data.data.user.id,
   265→   265→          email: data.data.user.email,
   266→   266→          name: data.data.user.name,
   267→   267→          avatarUrl: data.data.user.avatarUrl,
   268→   268→          isSuperAdmin: data.data.user.isSuperAdmin,
   269→   269→          preferredLocale: data.data.user.preferredLocale || 'en',
   270→   270→        },
   271→   271→        data.data.token
   272→   272→      );
   273→   273→      onOpenChange(false);
   274→   274→      setEmail('');
   275→   275→      setPassword('');
   276→   276→    } catch {
   277→   277→      setError(t('common.error', locale));
   278→   278→    } finally {
   279→   279→      setLoading(false);
   280→   280→    }
   281→   281→  };
   282→   282→
   283→   283→  return (
   284→   284→    <Dialog open={open} onOpenChange={onOpenChange}>
   285→   285→      <DialogContent className="sm:max-w-md">
   286→   286→        <DialogHeader>
   287→   287→          <DialogTitle className="text-xl">{t('auth.login', locale)}</DialogTitle>
   288→   288→          <DialogDescription>{t('brand.tagline', locale)}</DialogDescription>
   289→   289→        </DialogHeader>
   290→   290→        <form onSubmit={handleSubmit} className="space-y-4">
   291→   291→          {error && (
   292→   292→            <div className="flex items-center gap-2 text-sm text-destructive bg-destructive/10 p-3 rounded-md">
   293→   293→              <span className="shrink-0">⚠</span>
   294→   294→              {error}
   295→   295→            </div>
   296→   296→          )}
   297→   297→          <div className="space-y-2">
   298→   298→            <Label htmlFor="login-email">{t('auth.email', locale)}</Label>
   299→   299→            <Input id="login-email" type="email" placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} required autoComplete="email" />
   300→   300→          </div>
   301→   301→          <div className="space-y-2">
   302→   302→            <Label htmlFor="login-password">{t('auth.password', locale)}</Label>
   303→   303→            <div className="relative">
   304→   304→              <Input id="login-password" type={showPassword ? 'text' : 'password'} placeholder="••••••••" value={password} onChange={(e) => setPassword(e.target.value)} required minLength={8} autoComplete="current-password" className="pr-10" />
   305→   305→              <Button type="button" variant="ghost" size="icon" className="absolute right-0 top-0 h-full px-3 hover:bg-transparent" onClick={() => setShowPassword(!showPassword)}>
   306→   306→                {showPassword ? <EyeOff className="h-4 w-4 text-muted-foreground" /> : <Eye className="h-4 w-4 text-muted-foreground" />}
   307→   307→              </Button>
   308→   308→            </div>
   309→   309→          </div>
   310→   310→          <div className="flex items-center justify-between">
   311→   311→            <div className="flex items-center space-x-2">
   312→   312→              <Checkbox id="remember" checked={rememberMe} onCheckedChange={(v) => setRememberMe(v === true)} />
   313→   313→              <Label htmlFor="remember" className="text-sm text-muted-foreground cursor-pointer">{t('auth.rememberMe', locale)}</Label>
   314→   314→            </div>
   315→   315→            <Button variant="link" className="px-0 text-cyan-500" type="button">{t('auth.forgotPassword', locale)}</Button>
   316→   316→          </div>
   317→   317→          <Button type="submit" className="w-full bg-cyan-600 hover:bg-cyan-700 text-white" disabled={loading}>
   318→   318→            {loading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
   319→   319→            {t('auth.signIn', locale)}
   320→   320→          </Button>
   321→   321→        </form>
   322→   322→      </DialogContent>
   323→   323→    </Dialog>
   324→   324→  );
   325→   325→}

   329→   329→function RegisterDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   330→   330→  const { locale, setUser } = useAppStore();
   331→   331→  const [name, setName] = useState('');
   332→   332→  const [email, setEmail] = useState('');
   333→   333→  const [password, setPassword] = useState('');

   336→   336→  const [showPassword, setShowPassword] = useState(false);

   338→   338→  const [error, setError] = useState('');
   339→   339→
   340→   340→  const handleSubmit = async (e: React.FormEvent) => {
   341→   341→    e.preventDefault();
   342→   342→    setError('');
   343→   343→    if (password !== confirmPassword) {
   344→   344→      setError(locale === 'ar' ? 'كلمات المرور غير متطابقة' : 'Passwords do not match');

   350→   350→        method: 'POST',
   351→   351→        headers: { 'Content-Type': 'application/json' },
   352→   352→        body: JSON.stringify({ name, email, password, confirmPassword, locale: selectedLocale }),
   353→   353→      });
   354→   354→      const data = await res.json();
   355→   355→      if (!data.success) {
   356→   356→        setError(data.error || t('common.error', locale));
   357→   357→        return;
   358→   358→      }
   359→   359→      setUser(
   360→   360→        {
   361→   361→          id: data.data.user.id,
   362→   362→          email: data.data.user.email,
   363→   363→          name: data.data.user.name,
   364→   364→          avatarUrl: data.data.user.avatarUrl,
   365→   365→          isSuperAdmin: data.data.user.isSuperAdmin,
   366→   366→          preferredLocale: data.data.user.preferredLocale || selectedLocale,
   367→   367→        },
   368→   368→        data.data.token
   369→   369→      );
   370→   370→      onOpenChange(false);
   371→   371→      setName(''); setEmail(''); setPassword(''); setConfirmPassword('');
   372→   372→    } catch {
   373→   373→      setError(t('common.error', locale));
   374→   374→    } finally {
   375→   375→      setLoading(false);
   376→   376→    }
   377→   377→  };
