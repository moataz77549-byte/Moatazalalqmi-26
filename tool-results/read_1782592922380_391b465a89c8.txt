   236→// ─── Login Dialog ─────────────────────────────────────────────
   237→
   238→function LoginDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   239→  const { locale, setUser } = useAppStore();
   240→  const [email, setEmail] = useState('');
   241→  const [password, setPassword] = useState('');
   242→  const [showPassword, setShowPassword] = useState(false);
   243→  const [rememberMe, setRememberMe] = useState(false);
   244→  const [loading, setLoading] = useState(false);
   245→  const [error, setError] = useState('');
   246→
   247→  const handleSubmit = async (e: React.FormEvent) => {
   248→    e.preventDefault();
   249→    setError('');
   250→    setLoading(true);
   251→    try {
   252→      const res = await fetch('/api/v1/auth/login', {
   253→        method: 'POST',
   254→        headers: { 'Content-Type': 'application/json' },
   255→        body: JSON.stringify({ email, password, rememberMe }),
   256→      });
   257→      const data = await res.json();
   258→      if (!data.success) {
   259→        setError(data.error || t('common.error', locale));
   260→        return;
   261→      }
   262→      setUser(
   263→        {
   264→          id: data.data.user.id,
   265→          email: data.data.user.email,
   266→          name: data.data.user.name,
   267→          avatarUrl: data.data.user.avatarUrl,
   268→          isSuperAdmin: data.data.user.isSuperAdmin,
   269→          preferredLocale: data.data.user.preferredLocale || 'en',
   270→        },
   271→        data.data.token
   272→      );
   273→      onOpenChange(false);
   274→      setEmail('');
   275→      setPassword('');
   276→    } catch {
   277→      setError(t('common.error', locale));
   278→    } finally {
   279→      setLoading(false);
   280→    }
   281→  };
   282→
   283→  return (
   284→    <Dialog open={open} onOpenChange={onOpenChange}>
   285→      <DialogContent className="sm:max-w-md">
   286→        <DialogHeader>
   287→          <DialogTitle className="text-xl">{t('auth.login', locale)}</DialogTitle>
   288→          <DialogDescription>{t('brand.tagline', locale)}</DialogDescription>
   289→        </DialogHeader>
   290→        <form onSubmit={handleSubmit} className="space-y-4">
   291→          {error && (
   292→            <div className="flex items-center gap-2 text-sm text-destructive bg-destructive/10 p-3 rounded-md">
   293→              <span className="shrink-0">⚠</span>
   294→              {error}
   295→            </div>
   296→          )}
   297→          <div className="space-y-2">
   298→            <Label htmlFor="login-email">{t('auth.email', locale)}</Label>
   299→            <Input id="login-email" type="email" placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} required autoComplete="email" />
   300→          </div>
   301→          <div className="space-y-2">
   302→            <Label htmlFor="login-password">{t('auth.password', locale)}</Label>
   303→            <div className="relative">
   304→              <Input id="login-password" type={showPassword ? 'text' : 'password'} placeholder="••••••••" value={password} onChange={(e) => setPassword(e.target.value)} required minLength={8} autoComplete="current-password" className="pr-10" />
   305→              <Button type="button" variant="ghost" size="icon" className="absolute right-0 top-0 h-full px-3 hover:bg-transparent" onClick={() => setShowPassword(!showPassword)}>
   306→                {showPassword ? <EyeOff className="h-4 w-4 text-muted-foreground" /> : <Eye className="h-4 w-4 text-muted-foreground" />}
   307→              </Button>
   308→            </div>
   309→          </div>
   310→          <div className="flex items-center justify-between">
   311→            <div className="flex items-center space-x-2">
   312→              <Checkbox id="remember" checked={rememberMe} onCheckedChange={(v) => setRememberMe(v === true)} />
   313→              <Label htmlFor="remember" className="text-sm text-muted-foreground cursor-pointer">{t('auth.rememberMe', locale)}</Label>
   314→            </div>
   315→            <Button variant="link" className="px-0 text-cyan-500" type="button">{t('auth.forgotPassword', locale)}</Button>
   316→          </div>
   317→          <Button type="submit" className="w-full bg-cyan-600 hover:bg-cyan-700 text-white" disabled={loading}>
   318→            {loading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
   319→            {t('auth.signIn', locale)}
   320→          </Button>
   321→        </form>
   322→      </DialogContent>
   323→    </Dialog>
   324→  );
   325→}

   329→function RegisterDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   330→  const { locale, setUser } = useAppStore();
   331→  const [name, setName] = useState('');
   332→  const [email, setEmail] = useState('');
   333→  const [password, setPassword] = useState('');
   334→  const [confirmPassword, setConfirmPassword] = useState('');
   335→  const [selectedLocale, setSelectedLocale] = useState(locale);
   336→  const [showPassword, setShowPassword] = useState(false);
   337→  const [loading, setLoading] = useState(false);
   338→  const [error, setError] = useState('');
   339→
   340→  const handleSubmit = async (e: React.FormEvent) => {
   341→    e.preventDefault();
   342→    setError('');
   343→    if (password !== confirmPassword) {
   344→      setError(locale === 'ar' ? 'كلمات المرور غير متطابقة' : 'Passwords do not match');
   345→      return;
   346→    }
   347→    setLoading(true);
   348→    try {
   349→      const res = await fetch('/api/v1/auth/register', {
   350→        method: 'POST',
   351→        headers: { 'Content-Type': 'application/json' },
   352→        body: JSON.stringify({ name, email, password, confirmPassword, locale: selectedLocale }),
   353→      });
   354→      const data = await res.json();
   355→      if (!data.success) {
   356→        setError(data.error || t('common.error', locale));
   357→        return;
   358→      }
   359→      setUser(
   360→        {
   361→          id: data.data.user.id,
   362→          email: data.data.user.email,
   363→          name: data.data.user.name,
   364→          avatarUrl: data.data.user.avatarUrl,
   365→          isSuperAdmin: data.data.user.isSuperAdmin,
   366→          preferredLocale: data.data.user.preferredLocale || selectedLocale,
   367→        },
   368→        data.data.token
   369→      );
   370→      onOpenChange(false);
   371→      setName(''); setEmail(''); setPassword(''); setConfirmPassword('');
   372→    } catch {
   373→      setError(t('common.error', locale));
   374→    } finally {
   375→      setLoading(false);
   376→    }
   377→  };
   378→
   379→  return (
   380→    <Dialog open={open} onOpenChange={onOpenChange}>
   381→      <DialogContent className="sm:max-w-md">
   382→        <DialogHeader>
   383→          <DialogTitle className="text-xl">{t('auth.register', locale)}</DialogTitle>
   384→          <DialogDescription>{t('brand.tagline', locale)}</DialogDescription>
   385→        </DialogHeader>
   386→        <form onSubmit={handleSubmit} className="space-y-4">
   387→          {error && (
   388→            <div className="flex items-center gap-2 text-sm text-destructive bg-destructive/10 p-3 rounded-md">
   389→              <span className="shrink-0">⚠</span>
   390→              {error}
   391→            </div>
   392→          )}
   393→          <div className="space-y-2">
   394→            <Label htmlFor="reg-name">{t('auth.name', locale)}</Label>
   395→            <Input id="reg-name" placeholder="John Doe" value={name} onChange={(e) => setName(e.target.value)} required minLength={2} />
   396→          </div>
   397→          <div className="space-y-2">
   398→            <Label htmlFor="reg-email">{t('auth.email', locale)}</Label>
   399→            <Input id="reg-email" type="email" placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} required />
   400→          </div>
   401→          <div className="space-y-2">
   402→            <Label htmlFor="reg-password">{t('auth.password', locale)}</Label>
   403→            <div className="relative">
   404→              <Input id="reg-password" type={showPassword ? 'text' : 'password'} placeholder="••••••••" value={password} onChange={(e) => setPassword(e.target.value)} required minLength={8} className="pr-10" />
   405→              <Button type="button" variant="ghost" size="icon" className="absolute right-0 top-0 h-full px-3 hover:bg-transparent" onClick={() => setShowPassword(!showPassword)}>
   406→                {showPassword ? <EyeOff className="h-4 w-4 text-muted-foreground" /> : <Eye className="h-4 w-4 text-muted-foreground" />}

   408→            </div>
   409→            <p className="text-xs text-muted-foreground">{locale === 'ar' ? '8 أحرف على الأقل، يحتوي أحرف كبيرة وصغيرة وأرقام' : 'Min 8 chars with uppercase, lowercase, and number'}</p>
   410→          </div>
   411→          <div className="space-y-2">
   412→            <Label htmlFor="reg-confirm">{t('auth.confirmPassword', locale)}</Label>
   413→            <Input id="reg-confirm" type="password" placeholder="••••••••" value={confirmPassword} onChange={(e) => setConfirmPassword(e.target.value)} required />
   414→          </div>
   415→          <div className="space-y-2">
   416→            <Label>{t('auth.locale', locale)}</Label>

   514→function ChatView() {
   515→  const { locale } = useAppStore();
   516→  const [messages, setMessages] = useState<ChatMessage[]>([]);
   517→  const [input, setInput] = useState('');
   518→  const [provider, setProvider] = useState('openai');
   519→  const [model, setModel] = useState('gpt-4o');
   520→  const [sending, setSending] = useState(false);
   521→  const scrollRef = useRef<HTMLDivElement>(null);
   522→
   523→  const currentModels = MOCK_MODELS[provider] || [];
   524→
   525→  const handleProviderChange = useCallback((newProvider: string) => {
   526→    setProvider(newProvider);
   527→    const models = MOCK_MODELS[newProvider];
   528→    if (models && models.length > 0) {
   529→      setModel(models[0].value);
   530→    }
   531→  }, []);
   532→
   533→  useEffect(() => {
   534→    if (scrollRef.current) {
   535→      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
   536→    }
   537→  }, [messages]);
   538→
   539→  const handleSend = () => {
   540→    if (!input.trim() || sending) return;
   541→    const userMsg: ChatMessage = { id: Date.now().toString(), role: 'user', content: input.trim(), timestamp: new Date() };
   542→    setMessages((prev) => [...prev, userMsg]);
   543→    setInput('');
   544→    setSending(true);
   545→
   546→    setTimeout(() => {
   547→      const providerLabel = MOCK_PROVIDERS.find((p) => p.value === provider)?.label || provider;
   548→      const modelLabel = MOCK_MODELS[provider]?.find((m) => m.value === model)?.label || model;
   549→      const assistantMsg: ChatMessage = {
   550→        id: (Date.now() + 1).toString(),
   551→        role: 'assistant',
   552→        content: `I'm a simulated response from ${providerLabel} using ${modelLabel}. In production, this would connect to your configured AI provider and generate a real response based on your prompt: "${userMsg.content.substring(0, 100)}${userMsg.content.length > 100 ? '...' : ''}"`,
   553→        timestamp: new Date(),
   554→      };
   555→      setMessages((prev) => [...prev, assistantMsg]);
   556→      setSending(false);
   557→    }, 1200);
   558→  };
   559→
   560→  const handleKeyDown = (e: React.KeyboardEvent) => {
   561→    if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); handleSend(); }
   562→  };
   563→
   564→  return (
   565→    <div className="flex flex-col h-[calc(100vh-5rem)] animate-in fade-in duration-300">
   566→      {/* Chat Controls */}
   567→      <div className="flex flex-wrap items-center gap-3 pb-4">
   568→        <div className="flex items-center gap-2">
   569→          <Label className="text-sm whitespace-nowrap">{t('chat.provider', locale)}:</Label>
   570→          <Select value={provider} onValueChange={handleProviderChange}>
   571→            <SelectTrigger className="w-40"><SelectValue /></SelectTrigger>
   572→            <SelectContent>
   573→              {MOCK_PROVIDERS.map((p) => (<SelectItem key={p.value} value={p.value}>{p.label}</SelectItem>))}
   574→            </SelectContent>
   575→          </Select>
   576→        </div>
   577→        <div className="flex items-center gap-2">
   578→          <Label className="text-sm whitespace-nowrap">{t('chat.model', locale)}:</Label>
   579→          <Select value={model} onValueChange={setModel}>
   580→            <SelectTrigger className="w-48"><SelectValue /></SelectTrigger>
   581→            <SelectContent>
   582→              {currentModels.map((m) => (<SelectItem key={m.value} value={m.value}>{m.label}</SelectItem>))}
   583→            </SelectContent>
   584→          </Select>
   585→        </div>
   586→      </div>
   587→
   588→      {/* Messages Area */}
   589→      <Card className="flex-1 flex flex-col min-h-0">
   590→        <ScrollArea className="flex-1 p-4" ref={scrollRef}>
   591→          {messages.length === 0 ? (
   592→            <div className="flex flex-col items-center justify-center h-full min-h-[400px] text-center">
   593→              <Bot className="h-16 w-16 text-muted-foreground/30 mb-4" />

   595→              <p className="text-sm text-muted-foreground/60 mt-1 max-w-md">{t('chat.empty', locale)}</p>
   596→            </div>
   597→          ) : (
   598→            <div className="space-y-4 pb-4">
   599→              {messages.map((msg) => (
   600→                <div key={msg.id} className={cn('flex', msg.role === 'user' ? 'justify-end' : 'justify-start')}>
   601→                  <div className={cn('max-w-[80%] rounded-2xl px-4 py-3 text-sm leading-relaxed', msg.role === 'user' ? 'bg-cyan-600 text-white rounded-br-md' : 'bg-muted text-foreground rounded-bl-md')}>
   602→                    {msg.content}
   603→                  </div>
   604→                </div>
   605→              ))}
   606→              {sending && (
