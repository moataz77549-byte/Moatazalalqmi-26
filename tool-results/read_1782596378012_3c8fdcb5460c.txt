   254→function LoginDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   255→  const { locale, setUser } = useAppStore();
   256→  const [email, setEmail] = useState('');
   257→  const [password, setPassword] = useState('');
   258→  const [showPassword, setShowPassword] = useState(false);
   259→  const [rememberMe, setRememberMe] = useState(false);
   260→  const [loading, setLoading] = useState(false);
   261→  const [error, setError] = useState('');
   262→
   263→  const handleSubmit = async (e: React.FormEvent) => {
   264→    e.preventDefault();
   265→    setError('');
   266→    setLoading(true);
   267→    try {
   268→      const res = await fetch('/api/v1/auth/login', {
   269→        method: 'POST',
   270→        headers: { 'Content-Type': 'application/json' },
   271→        body: JSON.stringify({ email, password, rememberMe }),
   272→      });
   273→      const data = await res.json();
   274→      if (!data.success) {
   275→        setError(data.error || t('common.error', locale));
   276→        return;
   277→      }
   278→      setUser(
   279→        {
   280→          id: data.data.user.id,
   281→          email: data.data.user.email,
   282→          name: data.data.user.name,
   283→          avatarUrl: data.data.user.avatarUrl,
   284→          isSuperAdmin: data.data.user.isSuperAdmin,
   285→          preferredLocale: data.data.user.preferredLocale || 'en',
   286→        },
   287→        data.data.token
   288→      );
   289→      onOpenChange(false);
   290→      setEmail('');
   291→      setPassword('');
   292→    } catch {
   293→      setError(t('common.error', locale));
   294→    } finally {
   295→      setLoading(false);
   296→    }
   297→  };
   298→
   299→  return (
   300→    <Dialog open={open} onOpenChange={onOpenChange}>
   301→      <DialogContent className="sm:max-w-md">
   302→        <DialogHeader>
   303→          <DialogTitle className="text-xl">{t('auth.login', locale)}</DialogTitle>
   304→          <DialogDescription>{t('brand.tagline', locale)}</DialogDescription>
   305→        </DialogHeader>
   306→        <form onSubmit={handleSubmit} className="space-y-4">
   307→          {error && (
   308→            <div className="flex items-center gap-2 text-sm text-destructive bg-destructive/10 p-3 rounded-md">
   309→              <span className="shrink-0">⚠</span>
   310→              {error}
   311→            </div>
   312→          )}
   313→          <div className="space-y-2">
   314→            <Label htmlFor="login-email">{t('auth.email', locale)}</Label>
   315→            <Input id="login-email" type="email" placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} required autoComplete="email" />
   316→          </div>
   317→          <div className="space-y-2">
   318→            <Label htmlFor="login-password">{t('auth.password', locale)}</Label>
   319→            <div className="relative">
   320→              <Input id="login-password" type={showPassword ? 'text' : 'password'} placeholder="••••••••" value={password} onChange={(e) => setPassword(e.target.value)} required minLength={8} autoComplete="current-password" className="pr-10" />
   321→              <Button type="button" variant="ghost" size="icon" className="absolute right-0 top-0 h-full px-3 hover:bg-transparent" onClick={() => setShowPassword(!showPassword)}>
   322→                {showPassword ? <EyeOff className="h-4 w-4 text-muted-foreground" /> : <Eye className="h-4 w-4 text-muted-foreground" />}
   323→              </Button>
   324→            </div>
   325→          </div>
   326→          <div className="flex items-center justify-between">
   327→            <div className="flex items-center space-x-2">
   328→              <Checkbox id="remember" checked={rememberMe} onCheckedChange={(v) => setRememberMe(v === true)} />
   329→              <Label htmlFor="remember" className="text-sm text-muted-foreground cursor-pointer">{t('auth.rememberMe', locale)}</Label>
   330→            </div>
   331→            <Button variant="link" className="px-0 text-cyan-500" type="button">{t('auth.forgotPassword', locale)}</Button>
   332→          </div>
   333→          <Button type="submit" className="w-full bg-cyan-600 hover:bg-cyan-700 text-white" disabled={loading}>
   334→            {loading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
   335→            {t('auth.signIn', locale)}
   336→          </Button>
   337→        </form>
   338→      </DialogContent>
   339→    </Dialog>
   340→  );
   341→}

   345→function RegisterDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (o: boolean) => void }) {
   346→  const { locale, setUser } = useAppStore();
   347→  const [name, setName] = useState('');
   348→  const [email, setEmail] = useState('');
   349→  const [password, setPassword] = useState('');
   350→  const [confirmPassword, setConfirmPassword] = useState('');
   351→  const [selectedLocale, setSelectedLocale] = useState(locale);
   352→  const [showPassword, setShowPassword] = useState(false);
   353→  const [loading, setLoading] = useState(false);
   354→  const [error, setError] = useState('');
   355→
   356→  const handleSubmit = async (e: React.FormEvent) => {
   357→    e.preventDefault();
   358→    setError('');
   359→    if (password !== confirmPassword) {
   360→      setError(locale === 'ar' ? 'كلمات المرور غير متطابقة' : 'Passwords do not match');
   361→      return;
   362→    }
   363→    setLoading(true);
   364→    try {
   365→      const res = await fetch('/api/v1/auth/register', {
   366→        method: 'POST',
   367→        headers: { 'Content-Type': 'application/json' },
   368→        body: JSON.stringify({ name, email, password, confirmPassword, locale: selectedLocale }),
   369→      });
   370→      const data = await res.json();
   371→      if (!data.success) {
   372→        setError(data.error || t('common.error', locale));
   373→        return;
   374→      }
   375→      setUser(
   376→        {
   377→          id: data.data.user.id,
   378→          email: data.data.user.email,
   379→          name: data.data.user.name,
   380→          avatarUrl: data.data.user.avatarUrl,
   381→          isSuperAdmin: data.data.user.isSuperAdmin,
   382→          preferredLocale: data.data.user.preferredLocale || selectedLocale,
   383→        },
   384→        data.data.token
   385→      );
   386→      onOpenChange(false);
   387→      setName(''); setEmail(''); setPassword(''); setConfirmPassword('');
   388→    } catch {
   389→      setError(t('common.error', locale));
   390→    } finally {
   391→      setLoading(false);
   392→    }
   393→  };
   394→
   395→  return (
   396→    <Dialog open={open} onOpenChange={onOpenChange}>
   397→      <DialogContent className="sm:max-w-md">
   398→        <DialogHeader>
   399→          <DialogTitle className="text-xl">{t('auth.register', locale)}</DialogTitle>
   400→          <DialogDescription>{t('brand.tagline', locale)}</DialogDescription>
   401→        </DialogHeader>
   402→        <form onSubmit={handleSubmit} className="space-y-4">
   403→          {error && (
   404→            <div className="flex items-center gap-2 text-sm text-destructive bg-destructive/10 p-3 rounded-md">
   405→              <span className="shrink-0">⚠</span>
   406→              {error}
   407→            </div>
   408→          )}
   409→          <div className="space-y-2">
   410→            <Label htmlFor="reg-name">{t('auth.name', locale)}</Label>
   411→            <Input id="reg-name" placeholder="John Doe" value={name} onChange={(e) => setName(e.target.value)} required minLength={2} />
   412→          </div>
   413→          <div className="space-y-2">
   414→            <Label htmlFor="reg-email">{t('auth.email', locale)}</Label>
   415→            <Input id="reg-email" type="email" placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} required />
   416→          </div>
   417→          <div className="space-y-2">
   418→            <Label htmlFor="reg-password">{t('auth.password', locale)}</Label>
   419→            <div className="relative">
   420→              <Input id="reg-password" type={showPassword ? 'text' : 'password'} placeholder="••••••••" value={password} onChange={(e) => setPassword(e.target.value)} required minLength={8} className="pr-10" />
   421→              <Button type="button" variant="ghost" size="icon" className="absolute right-0 top-0 h-full px-3 hover:bg-transparent" onClick={() => setShowPassword(!showPassword)}>
   422→                {showPassword ? <EyeOff className="h-4 w-4 text-muted-foreground" /> : <Eye className="h-4 w-4 text-muted-foreground" />}
   423→              </Button>
   424→            </div>
   425→            <p className="text-xs text-muted-foreground">{locale === 'ar' ? '8 أحرف على الأقل، يحتوي أحرف كبيرة وصغيرة وأرقام' : 'Min 8 chars with uppercase, lowercase, and number'}</p>
   426→          </div>
   427→          <div className="space-y-2">
   428→            <Label htmlFor="reg-confirm">{t('auth.confirmPassword', locale)}</Label>
   429→            <Input id="reg-confirm" type="password" placeholder="••••••••" value={confirmPassword} onChange={(e) => setConfirmPassword(e.target.value)} required />
   430→          </div>
   431→          <div className="space-y-2">
   432→            <Label>{t('auth.locale', locale)}</Label>
   433→            <Select value={selectedLocale} onValueChange={(v) => setSelectedLocale(v as 'en' | 'ar')}>

   530→function ChatView() {
   531→  const { locale } = useAppStore();
   532→  const [messages, setMessages] = useState<ChatMessage[]>([]);
   533→  const [input, setInput] = useState('');
   534→  const [provider, setProvider] = useState('openai');

   536→  const [sending, setSending] = useState(false);

   538→
   539→  const currentModels = MOCK_MODELS[provider] || [];
   540→
   541→  const handleProviderChange = useCallback((newProvider: string) => {
   542→    setProvider(newProvider);
   543→    const models = MOCK_MODELS[newProvider];
   544→    if (models && models.length > 0) {
   545→      setModel(models[0].value);
   546→    }
   547→  }, []);
   548→
   549→  useEffect(() => {
   550→    if (scrollRef.current) {
   551→      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
   552→    }
   553→  }, [messages]);
   554→
   555→  const handleSend = () => {
   556→    if (!input.trim() || sending) return;
   557→    const userMsg: ChatMessage = { id: Date.now().toString(), role: 'user', content: input.trim(), timestamp: new Date() };
   558→    setMessages((prev) => [...prev, userMsg]);
   559→    setInput('');
   560→    setSending(true);
   561→
   562→    setTimeout(() => {
   563→      const providerLabel = MOCK_PROVIDERS.find((p) => p.value === provider)?.label || provider;
   564→      const modelLabel = MOCK_MODELS[provider]?.find((m) => m.value === model)?.label || model;
   565→      const assistantMsg: ChatMessage = {
   566→        id: (Date.now() + 1).toString(),
   567→        role: 'assistant',
   568→        content: `I'm a simulated response from ${providerLabel} using ${modelLabel}. In production, this would connect to your configured AI provider and generate a real response based on your prompt: "${userMsg.content.substring(0, 100)}${userMsg.content.length > 100 ? '...' : ''}"`,
   569→        timestamp: new Date(),
   570→      };
   571→      setMessages((prev) => [...prev, assistantMsg]);
   572→      setSending(false);
   573→    }, 1200);
   574→  };
   575→
   576→  const handleKeyDown = (e: React.KeyboardEvent) => {
   577→    if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); handleSend(); }
   578→  };
   579→
   580→  return (
   581→    <div className="flex flex-col h-[calc(100vh-5rem)] animate-in fade-in duration-300">
   582→      {/* Chat Controls */}
   583→      <div className="flex flex-wrap items-center gap-3 pb-4">

   596→            <SelectTrigger className="w-48"><SelectValue /></SelectTrigger>

   617→                  <div className={cn('max-w-[80%] rounded-2xl px-4 py-3 text-sm leading-relaxed', msg.role === 'user' ? 'bg-cyan-600 text-white rounded-br-md' : 'bg-muted text-foreground rounded-bl-md')}>
